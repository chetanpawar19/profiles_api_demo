#creating development server

-download and install Vagrant 
-open git bash inside project directory
-initialize vagrant file
git bash prompt>$ vagrant init ubuntu/bionic64

-copy code file from 
https://gist.github.com/LondonAppDev/199eef145a21587ea866b69d40d28682 and paste in vagrant file created at root directory
or copy code from I:\Computer Science\Vagrant\Vagrantfile_updated

-install virtual box 6.0.16
-control panel-turn windows features on or off-windows hypervisor platform
-f10-bios settings-enable windows hypervisor/viratualization
-open cmd as admin & type "bcdedit /set hypervisorlaunchtype off" & reboot 

git bash prompt>$ vagrant up
git bash prompt>$ vagrant ssh

#now guest ubuntu virtual machine development server is in use. 
-git bash prompt>$vagrant@ubuntu-bionic:~$ exit if you want to exit from guest ubuntu machine & point again to local machine dir

#to connect dev server again
-git bash prompt>$ vagrant ssh 
#navigate to vagrant directory
-vagrant@ubuntu-bionic:~$ cd /vagrant
#now virtual dev server & local project is in sync. try to create empty file using touch in virtual dev server. it will also get added in local project directory. 
-vagrant@ubuntu-bionic:/vagrant$ touch test.txt
#files in virtual dev server & local project dir shoult match
-vagrant@ubuntu-bionic:/vagrant$ ls
#if you delete any file in project.it will delete in virtual server as well.as sync is both side.

-create hello.py file in local dir
-vagrant@ubuntu-bionic:/vagrant$ ls
-vagrant@ubuntu-bionic:/vagrant$ python hello.py

#put files in git. open new git bash in local project dir
$ git add .
$ git commit -m "configured vagrant and hello.py"
$ git push origin
